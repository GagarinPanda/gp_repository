CREATE TABLE Users(
  userId SERIAL PRIMARY KEY,
  age INT
);

CREATE TABLE Items(
  itemId SERIAL PRIMARY KEY,
  price INT
);
  
CREATE TABLE Purchases(
  purchaseID SERIAL PRIMARY KEY,
  userId INT,
  itemId INT,
  date DATE,
  CONSTRAINT fk_user
  	FOREIGN KEY(userId)
  	REFERENCES Users(userId),
  CONSTRAINT fk_item
  	FOREIGN KEY(itemId)
  	REFERENCES Items(itemId)
);
